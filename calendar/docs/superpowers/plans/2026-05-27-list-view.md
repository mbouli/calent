# List View Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a List View mode to Calent with a re-arrangeable stickies board (75%) and a slim schedule column (25%), toggled from a Calendar / List switcher in the header.

**Architecture:** Top-level `appMode` state (`'calendar' | 'list'`) in `page.tsx` swaps the main content area between the existing calendar views and a new `ListView` component. `ListView` renders `StickiesPanel` (75%) and `SchedulePanel` (25%) side-by-side. Stickies have their own `useStickiesStore` hook backed by `localStorage`, mirroring `useCalendarStore`. Drag-to-reorder uses the HTML5 Drag and Drop API.

**Tech Stack:** Next.js 16 App Router, React 19, TypeScript, Tailwind v4, lucide-react, shadcn/ui (Dialog, Input, Button already installed)

```ts
'use client'

import { useState, useEffect, useCallback } from 'react'
import { Sticky, StickyItem, EventColor } from '../types'

const STORAGE_KEY = 'calent-stickies'

function makeId() { return crypto.randomUUID() }

function makeSampleStickies(): Sticky[] {
  return [
    {
      id: makeId(), title: 'MVP launch prep', color: 'indigo', order: 0, body: undefined,
      items: [
        { id: makeId(), text: 'Finalize designs', done: true },
        { id: makeId(), text: 'Write landing page copy', done: false },
        { id: makeId(), text: 'Send beta invites', done: false },
      ],
    },
    {
      id: makeId(), title: 'Groceries', color: 'rose', order: 1,
      body: 'Eggs, oat milk, sourdough, cherry tomatoes', items: [],
    },
    {
      id: makeId(), title: 'Home tasks', color: 'emerald', order: 2, body: undefined,
      items: [
        { id: makeId(), text: 'Call dentist', done: false },
        { id: makeId(), text: 'Pay rent', done: true },
      ],
    },
  ]
}

export function useStickiesStore() {
  const [stickies, setStickies] = useState<Sticky[]>([])

  useEffect(() => {
    try {
      const stored = localStorage.getItem(STORAGE_KEY)
      if (stored) {
        setStickies(JSON.parse(stored))
      } else {
        const samples = makeSampleStickies()
        setStickies(samples)
        localStorage.setItem(STORAGE_KEY, JSON.stringify(samples))
      }
    } catch {
      setStickies(makeSampleStickies())
    }
  }, [])

  const persist = useCallback((next: Sticky[]) => {
    setStickies(next)
    localStorage.setItem(STORAGE_KEY, JSON.stringify(next))
  }, [])

  const createSticky = useCallback(
    (title: string, color: EventColor) => {
      const s: Sticky = { id: makeId(), title, color, items: [], order: stickies.length }
      persist([...stickies, s])
    },
    [stickies, persist]
  )

  const updateSticky = useCallback(
    (id: string, updates: Partial<Omit<Sticky, 'id'>>) => {
      persist(stickies.map(s => s.id === id ? { ...s, ...updates } : s))
    },
    [stickies, persist]
  )

  const deleteSticky = useCallback(
    (id: string) => persist(stickies.filter(s => s.id !== id)),
    [stickies, persist]
  )

  const reorderStickies = useCallback(
    (ordered: Sticky[]) => persist(ordered.map((s, i) => ({ ...s, order: i }))),
    [persist]
  )

  const addItem = useCallback(
    (stickyId: string, text: string) => {
      persist(stickies.map(s =>
        s.id === stickyId
          ? { ...s, items: [...s.items, { id: makeId(), text, done: false }] }
          : s
      ))
    },
    [stickies, persist]
  )

  const toggleItem = useCallback(
    (stickyId: string, itemId: string) => {
      persist(stickies.map(s =>
        s.id === stickyId
          ? { ...s, items: s.items.map(it => it.id === itemId ? { ...it, done: !it.done } : it) }
          : s
      ))
    },
    [stickies, persist]
  )

  const clearDoneItems = useCallback(
    (stickyId: string) => {
      persist(stickies.map(s =>
        s.id === stickyId ? { ...s, items: s.items.filter(it => !it.done) } : s
      ))
    },
    [stickies, persist]
  )

  return {
    stickies: [...stickies].sort((a, b) => a.order - b.order),
    createSticky,
    updateSticky,
    deleteSticky,
    reorderStickies,
    addItem,
    toggleItem,
    clearDoneItems,
  }
}
```

```tsx
'use client'

import { useState } from 'react'
import { Plus } from 'lucide-react'
import { cn } from '@/lib/utils'
import { Sticky, EventColor } from '../types'
import { StickyCard } from './StickyCard'
import { StickyModal } from './StickyModal'

interface StickysPanelProps {
  stickies: Sticky[]
  onCreateSticky: (title: string, color: EventColor) => void
  onUpdateSticky: (id: string, updates: Partial<Omit<Sticky, 'id'>>) => void
  onDeleteSticky: (id: string) => void
  onReorderStickies: (ordered: Sticky[]) => void
  onAddItem: (stickyId: string, text: string) => void
  onToggleItem: (stickyId: string, itemId: string) => void
  onClearDoneItems: (stickyId: string) => void
}

export function StickiesPanel({
  stickies,
  onCreateSticky,
  onUpdateSticky,
  onDeleteSticky,
  onReorderStickies,
  onAddItem,
  onToggleItem,
  onClearDoneItems,
}: StickysPanelProps) {
  const [showModal, setShowModal] = useState(false)
  const [dragId, setDragId] = useState<string | null>(null)

  const handleDragStart = (e: React.DragEvent, id: string) => {
    setDragId(id)
    e.dataTransfer.effectAllowed = 'move'
  }

  const handleDragOver = (e: React.DragEvent) => {
    e.preventDefault()
    e.dataTransfer.dropEffect = 'move'
  }

  const handleDrop = (e: React.DragEvent, targetId: string) => {
    e.preventDefault()
    if (!dragId || dragId === targetId) { setDragId(null); return }
    const list = [...stickies]
    const fromIdx = list.findIndex(s => s.id === dragId)
    const toIdx = list.findIndex(s => s.id === targetId)
    const [moved] = list.splice(fromIdx, 1)
    list.splice(toIdx, 0, moved)
    onReorderStickies(list)
    setDragId(null)
  }

  return (
    <>
      <div className="flex items-center justify-between px-4 py-2.5 border-b border-border/30 bg-background shrink-0">
        <span className="text-[10px] font-semibold uppercase tracking-[0.15em] text-muted-foreground/50">
          Stickies
        </span>
        <button
          onClick={() => setShowModal(true)}
          className="flex items-center gap-1 text-[10px] font-medium text-muted-foreground/60 hover:text-foreground transition-colors"
        >
          <Plus size={12} />
          New
        </button>
      </div>

      <div className="flex-1 overflow-y-auto p-4">
        <div className="grid grid-cols-3 gap-3 content-start">
          {stickies.map(sticky => (
            <StickyCard
              key={sticky.id}
              sticky={sticky}
              onUpdateTitle={title => onUpdateSticky(sticky.id, { title })}
              onUpdateBody={body => onUpdateSticky(sticky.id, { body })}
              onToggleItem={itemId => onToggleItem(sticky.id, itemId)}
              onAddItem={text => onAddItem(sticky.id, text)}
              onClearDone={() => onClearDoneItems(sticky.id)}
              onChangeColor={color => onUpdateSticky(sticky.id, { color })}
              onDelete={() => onDeleteSticky(sticky.id)}
              onDragStart={e => handleDragStart(e, sticky.id)}
              onDragOver={handleDragOver}
              onDrop={e => handleDrop(e, sticky.id)}
            />
          ))}

          {/* New sticky placeholder card */}
          <button
            onClick={() => setShowModal(true)}
            className={cn(
              'rounded-xl border-2 border-dashed border-border/40 min-h-[80px]',
              'flex flex-col items-center justify-center gap-1.5',
              'text-muted-foreground/40 hover:text-muted-foreground/60 hover:border-border/60',
              'transition-colors'
            )}
          >
            <Plus size={16} />
            <span className="text-[10px] font-medium">New sticky</span>
          </button>
        </div>
      </div>

      <StickyModal
        open={showModal}
        onSave={onCreateSticky}
        onClose={() => setShowModal(false)}
      />
    </>
  )
}
```

```tsx
import { isSameDay, isSameMonth, getWeekDays, startOfWeek, getMonthGrid, formatTime } from '../lib/dateUtils'

const DOT: Record<EventColor, string> = {
  indigo: 'bg-indigo-400', violet: 'bg-violet-400', rose: 'bg-rose-400',
  emerald: 'bg-emerald-400', amber: 'bg-amber-400', sky: 'bg-sky-400',
}

function getDays(currentDate: Date, viewMode: ViewMode): Date[] {
  if (viewMode === 'day') return [currentDate]
  if (viewMode === 'week') return getWeekDays(startOfWeek(currentDate))
  // month: all non-null days in the month grid
  return getMonthGrid(currentDate).filter((d): d is Date => d !== null)
}

interface SchedulePanelProps {
  events: CalendarEvent[]
  currentDate: Date
  viewMode: ViewMode
  onEditEvent: (event: CalendarEvent) => void
}

export function SchedulePanel({ events, currentDate, viewMode, onEditEvent }: SchedulePanelProps) {
  const today = new Date()
  const days = getDays(currentDate, viewMode)

  const eventsForDay = (day: Date) =>
    events
      .filter(e => isSameDay(e.start, day))
      .sort((a, b) => a.start.getTime() - b.start.getTime())

  return (
    <>
      <div className="px-4 py-2.5 border-b border-border/30 bg-background shrink-0">
        <span className="text-[10px] font-semibold uppercase tracking-[0.15em] text-muted-foreground/50">
          Schedule
        </span>
      </div>

      <div className="flex-1 overflow-y-auto">
        {days.map((day, i) => {
          const dayEvents = eventsForDay(day)
          const isToday = isSameDay(day, today)
          return (
            <div key={i} className="px-3 py-2 border-b border-border/20">
              {/* Day label */}
              <div className="flex items-center gap-1.5 mb-1.5">
                <span className="text-[9px] font-semibold uppercase tracking-wide text-muted-foreground/40">
                  {day.toLocaleDateString('en-US', { weekday: 'short' })}
                </span>
                <span className={cn(
                  'text-[11px] font-bold leading-none flex items-center justify-center',
                  isToday && 'bg-gray-900 text-white w-[18px] h-[18px] rounded-[4px]',
                  !isToday && 'text-gray-700'
                )}>
                  {day.getDate()}
                </span>
              </div>

              {/* Events */}
              {dayEvents.length === 0 ? (
                <p className="text-[9px] text-muted-foreground/25 italic">No events</p>
              ) : (
                <div className="space-y-1">
                  {dayEvents.map(ev => (
                    <button
                      key={ev.id}
                      onClick={() => onEditEvent(ev)}
                      className="w-full flex items-start gap-1.5 text-left group/ev"
                    >
                      <div className={cn('w-1.5 h-1.5 rounded-full mt-1 flex-shrink-0', DOT[ev.color])} />
                      <div className="flex-1 min-w-0">
                        <p className="text-[10px] font-medium text-gray-700 truncate group-hover/ev:text-gray-900 transition-colors">
                          {ev.title}
                        </p>
                        <p className="text-[9px] text-muted-foreground/40">
                          {formatTime(ev.start)}–{formatTime(ev.end)}
                        </p>
                      </div>
                    </button>
                  ))}
                </div>
              )}
            </div>
          )
        })}
      </div>
    </>
  )
}
```

```tsx
'use client'

import { useState, useRef } from 'react'
import { CalendarHeader } from './components/CalendarHeader'
import { TimeGrid } from './components/TimeGrid'
import { MonthView } from './components/MonthView'
import { MiniCalendar } from './components/MiniCalendar'
import { ListView } from './components/ListView'
import { EventModal } from './components/EventModal'
import { useCalendarStore } from './hooks/useCalendarStore'
import { CalendarEvent } from './types'
import { startOfWeek } from './lib/dateUtils'

interface ModalState {
  open: boolean
  event?: CalendarEvent
  defaultStart?: Date
  defaultEnd?: Date
}

export default function HomePage() {
  const {
    events,
    currentDate,
    setCurrentDate,
    viewMode,
    setViewMode,
    createEvent,
    updateEvent,
    deleteEvent,
  } = useCalendarStore()

  const [modal, setModal] = useState<ModalState>({ open: false })
  const [appMode, setAppMode] = useState<'calendar' | 'list'>('calendar')
  const navDirectionRef = useRef(0)

  const handleDateChange = (newDate: Date) => {
    navDirectionRef.current = newDate > currentDate ? 1 : -1
    setCurrentDate(newDate)
  }

  const handleViewModeChange = (mode: typeof viewMode) => {
    navDirectionRef.current = 0
    setViewMode(mode)
  }

  const animKey =
    viewMode === 'week'
      ? `week-${startOfWeek(currentDate).getTime()}`
      : viewMode === 'day'
      ? `day-${currentDate.toDateString()}`
      : `month-${currentDate.getFullYear()}-${currentDate.getMonth()}`

  const openCreate = (start: Date, end: Date) =>
    setModal({ open: true, defaultStart: start, defaultEnd: end })

  const openEdit = (event: CalendarEvent) =>
    setModal({ open: true, event })

  const handleSave = (data: Omit<CalendarEvent, 'id'>) => {
    if (modal.event) {
      updateEvent(modal.event.id, data)
    } else {
      createEvent(data)
    }
    setModal({ open: false })
  }

  const handleDelete = () => {
    if (modal.event) deleteEvent(modal.event.id)
    setModal({ open: false })
  }

  const handleMiniCalendarSelect = (date: Date) => {
    navDirectionRef.current = date > currentDate ? 1 : -1
    setCurrentDate(date)
    if (viewMode === 'month') setViewMode('day')
  }

  const handleMonthDayClick = (date: Date) => {
    setCurrentDate(date)
    setViewMode('day')
  }

  return (
    <div className="flex flex-col h-screen overflow-hidden bg-background">
      <CalendarHeader
        currentDate={currentDate}
        viewMode={viewMode}
        onDateChange={handleDateChange}
        onViewModeChange={handleViewModeChange}
        appMode={appMode}
        onAppModeChange={setAppMode}
      />

      <div className="flex flex-1 overflow-hidden">
        {appMode === 'calendar' && (
          <MiniCalendar
            selectedDate={currentDate}
            viewMode={viewMode}
            onSelectDate={handleMiniCalendarSelect}
          />
        )}

        {appMode === 'list' ? (
          <ListView
            events={events}
            currentDate={currentDate}
            viewMode={viewMode}
            onEditEvent={openEdit}
          />
        ) : viewMode === 'month' ? (
          <MonthView
            currentDate={currentDate}
            events={events}
            onDayClick={handleMonthDayClick}
            onEventClick={openEdit}
          />
        ) : (
          <TimeGrid
            events={events}
            currentDate={currentDate}
            viewMode={viewMode}
            navDirection={navDirectionRef.current}
            animKey={animKey}
            onCreateEvent={openCreate}
            onEditEvent={openEdit}
            onUpdateEvent={updateEvent}
            onDeleteEvent={deleteEvent}
          />
        )}
      </div>

      <EventModal
        open={modal.open}
        event={modal.event}
        defaultStart={modal.defaultStart}
        defaultEnd={modal.defaultEnd}
        onSave={handleSave}
        onDelete={modal.event ? handleDelete : undefined}
        onClose={() => setModal({ open: false })}
      />
    </div>
  )
}
```
